  197. 197                             if (!$this->_checkConfigForFeatures($codeFeature, $attr)) // feature missing in config.xml file, or wrong attribute value

  204. 204             $this->_errors[] .= Tools::displayError('config.xml theme file has not been created for this version of prestashop.');

  206. 206         }

  208. 208     }
