  247. 247     private function watermarkByImage($imagepath, $watermarkpath, $outputpath)
  248. 248     {    
  249. 249         $Xoffset = $Yoffset = $xpos = $ypos = 0;
  250. 250         if (!$image = imagecreatefromjpeg($imagepath))
  251. 251             return false;
  252. 252         if (!$imagew = imagecreatefromgif ($watermarkpath))
  253. 253             die ($this->l('The watermark image is not a real gif, please CONVERT the image.'));
  254. 254         list($watermarkWidth, $watermarkHeight) = getimagesize($watermarkpath); 
  255. 255         list($imageWidth, $imageHeight) = getimagesize($imagepath); 
  256. 256         if ($this->xAlign == "middle") { $xpos = $imageWidth/2 - $watermarkWidth/2 + $Xoffset; } 
  257. 257         if ($this->xAlign == "left") { $xpos = 0 + $Xoffset; } 
  258. 258         if ($this->xAlign == "right") { $xpos = $imageWidth - $watermarkWidth - $Xoffset; } 
  259. 259         if ($this->yAlign == "middle") { $ypos = $imageHeight/2 - $watermarkHeight/2 + $Yoffset; } 
  260. 260         if ($this->yAlign == "top") { $ypos = 0 + $Yoffset; } 
  261. 261         if ($this->yAlign == "bottom") { $ypos = $imageHeight - $watermarkHeight - $Yoffset; } 
  262. 262         if (!imagecopymerge($image, $imagew, $xpos, $ypos, 0, 0, $watermarkWidth, $watermarkHeight, $this->transparency))
  263. 263             return false;
  264. 264         return imagejpeg($image, $outputpath, 100); 
  265. 265     }
