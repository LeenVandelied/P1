  152. 152     public function hookPayment($params)
  153. 153     {
  154. 154         global $cookie, $smarty;
  155. 155 
  156. 156         if (Configuration::get('PS_SSL_ENABLED') || (!empty($_SERVER['HTTPS']) && strtolower($_SERVER['HTTPS']) != 'off'))
  157. 157         {
  158. 158             $invoiceAddress = new Address((int)$params['cart']->id_address_invoice);
  159. 159             $customer = new Customer((int)$cookie->id_customer);
  160. 160             
  161. 161             $authorizeAIMParams = array();
  162. 162             $authorizeAIMParams['x_login'] = Configuration::get('AUTHORIZE_AIM_LOGIN_ID');
  163. 163             $authorizeAIMParams['x_tran_key'] = Configuration::get('AUTHORIZE_AIM_KEY');
  164. 164             $authorizeAIMParams['x_version'] = '3.1';
  165. 165             $authorizeAIMParams['x_delim_data'] = 'TRUE';
  166. 166             $authorizeAIMParams['x_delim_char'] = '|';
  167. 167             $authorizeAIMParams['x_relay_response'] = 'FALSE';
  168. 168             $authorizeAIMParams['x_type'] = 'AUTH_CAPTURE';
  169. 169             $authorizeAIMParams['x_method'] = 'CC';
  170. 170             $authorizeAIMParams['x_test_request'] = Configuration::get('AUTHORIZE_AIM_DEMO');
  171. 171             $authorizeAIMParams['x_invoice_num'] = (int)$params['cart']->id;
  172. 172             $authorizeAIMParams['x_amount'] = number_format($params['cart']->getOrderTotal(true, 3), 2, '.', '');
  173. 173             $authorizeAIMParams['x_address'] = $invoiceAddress->address1.' '.$invoiceAddress->address2;
  174. 174             $authorizeAIMParams['x_zip'] = $invoiceAddress->postcode;
  175. 175             $authorizeAIMParams['x_first_name'] = $customer->firstname;
  176. 176             $authorizeAIMParams['x_last_name'] = $customer->lastname;
  177. 177             
  178. 178             $isFailed = Tools::getValue('aimerror');
  179. 179 
  180. 180             $cards = array();
  181. 181             $cards['visa'] = Configuration::get('AUTHORIZE_AIM_CARD_VISA') == 'on' ? 1 : 0;

  185. 185 
  186. 186             $smarty->assign('p', $authorizeAIMParams);
