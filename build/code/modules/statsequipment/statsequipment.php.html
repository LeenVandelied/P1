  56.  56     private function getEquipment()
  57.  57     {
  58.  58         $result = Db::getInstance(_PS_USE_SQL_SLAVE_)->ExecuteS('
  59.  59         SELECT DISTINCT g.*
  60.  60         FROM `'._DB_PREFIX_.'connections` c 
  61.  61         LEFT JOIN `'._DB_PREFIX_.'guest` g ON g.`id_guest` = c.`id_guest`

  63.  63         
  64.  64         $calcArray = array('jsOK' => 0, 'jsKO' => 0, 'javaOK' => 0, 'javaKO' => 0, 'wmpOK' => 0, 'wmpKO' => 0, 'qtOK' => 0, 'qtKO' => 0, 'realOK' => 0, 'realKO' => 0, 'flashOK' => 0, 'flashKO' => 0, 'directorOK' => 0, 'directorKO' => 0);
  65.  65         while ($row = Db::getInstance(_PS_USE_SQL_SLAVE_)->nextRow($result))
  66.  66         {
  67.  67             if (!$row['javascript'])
  68.  68             {
  69.  69                 ++$calcArray['jsKO'];
  70.  70                 continue;
  71.  71             }
  72.  72             ++$calcArray['jsOK'];
  73.  73             ($row['windows_media']) ? ++$calcArray['wmpOK'] : ++$calcArray['wmpKO'];
  74.  74             ($row['real_player']) ? ++$calcArray['realOK'] : ++$calcArray['realKO'];
  75.  75             ($row['adobe_flash']) ? ++$calcArray['flashOK'] : ++$calcArray['flashKO'];
  76.  76             ($row['adobe_director']) ? ++$calcArray['directorOK'] : ++$calcArray['directorKO'];
  77.  77             ($row['sun_java']) ? ++$calcArray['javaOK'] : ++$calcArray['javaKO'];
  78.  78             ($row['apple_quicktime']) ? ++$calcArray['qtOK'] : ++$calcArray['qtKO'];
  79.  79         }
  80.  80         
  81.  81         if (!$calcArray['jsOK'])
  82.  82             return false;
  83.  83             
  84.  84         $equip = array(
  85.  85             'Windows Media Player' => $calcArray['wmpOK'] / ($calcArray['wmpOK'] + $calcArray['wmpKO']),
  86.  86             'Real Player' => $calcArray['realOK'] / ($calcArray['realOK'] + $calcArray['realKO']),

  89.  89             'Adobe Flash' => $calcArray['flashOK'] / ($calcArray['flashOK'] + $calcArray['flashKO']),
  90.  90             'Adobe Shockwave' => $calcArray['directorOK'] / ($calcArray['directorOK'] + $calcArray['directorKO'])
  91.  91         );
  92.  92         arsort($equip);
  93.  93         return $equip;
  94.  94     }
