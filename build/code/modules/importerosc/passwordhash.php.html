  38.  38         if ($iteration_count_log2 < 4 || $iteration_count_log2 > 31)
  39.  39             $iteration_count_log2 = 8;

  49.  49     function get_random_bytes($count)
  50.  50     {
  51.  51         $output = '';
  52.  52         if (@is_readable('/dev/urandom') &&
  53.  53             ($fh = @fopen('/dev/urandom', 'rb'))) {
  54.  54             $output = fread($fh, $count);
  55.  55             fclose($fh);
  56.  56         }
  57.  57 
  58.  58         if (strlen($output) < $count) {
  59.  59             $output = '';
  60.  60             for ($i = 0; $i < $count; $i += 16) {
  61.  61                 $this->random_state =
  62.  62                     md5(microtime() . $this->random_state);
  63.  63                 $output .=
  64.  64                     pack('H*', md5($this->random_state));
  65.  65             }
  66.  66             $output = substr($output, 0, $count);
  67.  67         }

  72.  72     function encode64($input, $count)
  73.  73     {
  74.  74         $output = '';
  75.  75         $i = 0;
  76.  76         do {
  77.  77             $value = ord($input[$i++]);
  78.  78             $output .= $this->itoa64[$value & 0x3f];
  79.  79             if ($i < $count)
  80.  80                 $value |= ord($input[$i]) << 8;
  81.  81             $output .= $this->itoa64[($value >> 6) & 0x3f];
  82.  82             if ($i++ >= $count)
  83.  83                 break;
  84.  84             if ($i < $count)
  85.  85                 $value |= ord($input[$i]) << 16;
  86.  86             $output .= $this->itoa64[($value >> 12) & 0x3f];
  87.  87             if ($i++ >= $count)
  88.  88                 break;
  89.  89             $output .= $this->itoa64[($value >> 18) & 0x3f];
  90.  90         } while ($i < $count);

  105. 105     function crypt_private($password, $setting)
  106. 106     {
  107. 107         $output = '*0';
  108. 108         if (substr($setting, 0, 2) == $output)
  109. 109             $output = '*1';
  110. 110 
  111. 111         $id = substr($setting, 0, 3);
  112. 112         # We use "$P$", phpBB3 uses "$H$" for the same thing
  113. 113         if ($id != '$P$' && $id != '$H$')
  114. 114             return $output;
  115. 115 
  116. 116         $count_log2 = strpos($this->itoa64, $setting[3]);
  117. 117         if ($count_log2 < 7 || $count_log2 > 30)
  118. 118             return $output;
  119. 119 
  120. 120         $count = 1 << $count_log2;
  121. 121 
  122. 122         $salt = substr($setting, 4, 8);
  123. 123         if (strlen($salt) != 8)
  124. 124             return $output;
  125. 125 
  126. 126         # We're kind of forced to use MD5 here since it's the only
  127. 127         # cryptographic primitive available in all versions of PHP
  128. 128         # currently in use.  To implement our own low-level crypto
  129. 129         # in PHP would result in much worse performance and
  130. 130         # consequently in lower iteration counts and hashes that are
  131. 131         # quicker to crack (by non-PHP code).
  132. 132         if (PHP_VERSION >= '5') {
  133. 133             $hash = md5($salt . $password, TRUE);
  134. 134             do {
  135. 135                 $hash = md5($hash . $password, TRUE);
  136. 136             } while (--$count);
  137. 137         } else {
  138. 138             $hash = pack('H*', md5($salt . $password));
  139. 139             do {
  140. 140                 $hash = pack('H*', md5($hash . $password));
  141. 141             } while (--$count);
  142. 142         }
  143. 143 
  144. 144         $output = substr($setting, 0, 12);
  145. 145         $output .= $this->encode64($hash, 16);
  146. 146 
  147. 147         return $output;
  148. 148     }

  150. 150     function gensalt_extended($input)
  151. 151     {
  152. 152         $count_log2 = min($this->iteration_count_log2 + 8, 24);
  153. 153         # This should be odd to not reveal weak DES keys, and the
  154. 154         # maximum valid value is (2**24 - 1) which is odd anyway.
  155. 155         $count = (1 << $count_log2) - 1;
  156. 156 
  157. 157         $output = '_';
  158. 158         $output .= $this->itoa64[$count & 0x3f];
  159. 159         $output .= $this->itoa64[($count >> 6) & 0x3f];
  160. 160         $output .= $this->itoa64[($count >> 12) & 0x3f];
  161. 161         $output .= $this->itoa64[($count >> 18) & 0x3f];
  162. 162 
  163. 163         $output .= $this->encode64($input, 3);
  164. 164 
  165. 165         return $output;
  166. 166     }

  168. 168     function gensalt_blowfish($input)
  169. 169     {
  170. 170         # This one needs to use a different order of characters and a
  171. 171         # different encoding scheme from the one in encode64() above.
  172. 172         # We care because the last character in our encoded string will
  173. 173         # only represent 2 bits.  While two known implementations of
  174. 174         # bcrypt will happily accept and correct a salt string which
  175. 175         # has the 4 unused bits set to non-zero, we do not want to take
  176. 176         # chances and we also do not want to waste an additional byte
  177. 177         # of entropy.
  178. 178         $itoa64 = './ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  179. 179 
  180. 180         $output = '$2a$';
  181. 181         $output .= chr(ord('0') + $this->iteration_count_log2 / 10);
  182. 182         $output .= chr(ord('0') + $this->iteration_count_log2 % 10);
  183. 183         $output .= '$';
  184. 184 
  185. 185         $i = 0;
  186. 186         do {
  187. 187             $c1 = ord($input[$i++]);
  188. 188             $output .= $itoa64[$c1 >> 2];
  189. 189             $c1 = ($c1 & 0x03) << 4;
  190. 190             if ($i >= 16) {
  191. 191                 $output .= $itoa64[$c1];
  192. 192                 break;
  193. 193             }
  194. 194 
  195. 195             $c2 = ord($input[$i++]);
  196. 196             $c1 |= $c2 >> 4;
  197. 197             $output .= $itoa64[$c1];
  198. 198             $c1 = ($c2 & 0x0f) << 2;
  199. 199 
  200. 200             $c2 = ord($input[$i++]);
  201. 201             $c1 |= $c2 >> 6;
  202. 202             $output .= $itoa64[$c1];
  203. 203             $output .= $itoa64[$c2 & 0x3f];
  204. 204         } while (1);
  205. 205 
  206. 206         return $output;
  207. 207     }

  209. 209     function HashPassword($password)
  210. 210     {
  211. 211         $random = '';
  212. 212 
  213. 213         if (CRYPT_BLOWFISH == 1 && !$this->portable_hashes) {
  214. 214             $random = $this->get_random_bytes(16);
  215. 215             $hash =
  216. 216                 crypt($password, $this->gensalt_blowfish($random));
  217. 217             if (strlen($hash) == 60)
  218. 218                 return $hash;
  219. 219         }
  220. 220 
  221. 221         if (CRYPT_EXT_DES == 1 && !$this->portable_hashes) {
  222. 222             if (strlen($random) < 3)
  223. 223                 $random = $this->get_random_bytes(3);
  224. 224             $hash =
  225. 225                 crypt($password, $this->gensalt_extended($random));
  226. 226             if (strlen($hash) == 20)
  227. 227                 return $hash;
  228. 228         }
  229. 229 
  230. 230         if (strlen($random) < 6)
  231. 231             $random = $this->get_random_bytes(6);
  232. 232         $hash =
  233. 233             $this->crypt_private($password,
  234. 234             $this->gensalt_private($random));
  235. 235         if (strlen($hash) == 34)
  236. 236             return $hash;
  237. 237 
  238. 238         # Returning '*' on error is safe here, but would _not_ be safe
  239. 239         # in a crypt(3)-like function used _both_ for generating new
  240. 240         # hashes and for validating passwords against existing hashes.
  241. 241         return '*';
  242. 242     }
