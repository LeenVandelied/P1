  73.  73     public function changeIdOrderState($new_order_state = NULL, $id_order)
  74.  74     {
  75.  75         if ($new_order_state != NULL)
  76.  76         {
  77.  77             Hook::updateOrderStatus((int)($new_order_state), (int)($id_order));
  78.  78             $order = new Order((int)($id_order));
  79.  79             
  80.  80             /* Best sellers */
  81.  81             $newOS = new OrderState((int)($new_order_state), $order->id_lang);
  82.  82             $oldOrderStatus = OrderHistory::getLastOrderState((int)($id_order));
  83.  83             $cart = Cart::getCartByOrderId($id_order);
  84.  84             $isValidated = $this->isValidated();
  85.  85             if (Validate::isLoadedObject($cart))
  86.  86                 foreach ($cart->getProducts() as $product)
  87.  87                 {
  88.  88                     /* If becoming logable => adding sale */
  89.  89                     if ($newOS->logable AND (!$oldOrderStatus OR !$oldOrderStatus->logable))
  90.  90                         ProductSale::addProductSale($product['id_product'], $product['cart_quantity']);
  91.  91                     /* If becoming unlogable => removing sale */
  92.  92                     elseif (!$newOS->logable AND ($oldOrderStatus AND $oldOrderStatus->logable))
  93.  93                         ProductSale::removeProductSale($product['id_product'], $product['cart_quantity']);
  94.  94                     if (!$isValidated AND $newOS->logable AND isset($oldOrderStatus) AND $oldOrderStatus AND $oldOrderStatus->id == Configuration::get('PS_OS_ERROR'))
  95.  95                     {
  96.  96                         Product::updateQuantity($product);
  97.  97                     }
  98.  98                 }
  99.  99             
  100. 100             $this->id_order_state = (int)($new_order_state);
  101. 101             
  102. 102             /* Change invoice number of order ? */
  103. 103             if (!Validate::isLoadedObject($newOS) OR !Validate::isLoadedObject($order))
  104. 104                 die(Tools::displayError('Invalid new order state'));
  105. 105             
  106. 106             /* The order is valid only if the invoice is available and the order is not cancelled */
  107. 107             $order->valid = $newOS->logable;
  108. 108             $order->update();
  109. 109 
  110. 110             if ($newOS->invoice AND !$order->invoice_number)
  111. 111                 $order->setInvoice();
  112. 112             if ($newOS->delivery AND !$order->delivery_number)
  113. 113                 $order->setDelivery();
  114. 114             Hook::postUpdateOrderStatus((int)($new_order_state), (int)($id_order));
  115. 115         }
  116. 116     }
