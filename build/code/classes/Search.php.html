  97.  97     public static function sanitize($string, $id_lang, $indexation = false)
  98.  98     {
  99.  99         $string = Tools::strtolower(strip_tags($string));
  100. 100         $string = html_entity_decode($string, ENT_NOQUOTES, 'utf-8');
  101. 101 
  102. 102         $string = preg_replace('/(['.PREG_CLASS_NUMBERS.']+)['.PREG_CLASS_PUNCTUATION.']+(?=['.PREG_CLASS_NUMBERS.'])/u', '\1', $string);
  103. 103         $string = preg_replace('/['.PREG_CLASS_SEARCH_EXCLUDE.']+/u', ' ', $string);
  104. 104 
  105. 105         if ($indexation)
  106. 106             $string = preg_replace('/[._-]+/', '', $string);
  107. 107         else
  108. 108         {
  109. 109             $string = preg_replace('/[._]+/', '', $string);
  110. 110             $string = ltrim(preg_replace('/([^ ])-/', '$1', ' '.$string));
  111. 111             $string = preg_replace('/[._]+/', '', $string);
  112. 112             $string = preg_replace('/[^\s]-+/', '', $string);
  113. 113         }
  114. 114 
  115. 115         $blacklist = Configuration::get('PS_SEARCH_BLACKLIST', (int)$id_lang);
  116. 116         if (!empty($blacklist))
  117. 117         {
  118. 118             $string = preg_replace('/(?<=\s)('.$blacklist.')(?=\s)/Su', '', $string);
  119. 119             $string = preg_replace('/^('.$blacklist.')(?=\s)/Su', '', $string);
  120. 120             $string = preg_replace('/(?<=\s)('.$blacklist.')$/Su', '', $string);
  121. 121             $string = preg_replace('/^('.$blacklist.')$/Su', '', $string);
  122. 122         }
  123. 123 
  124. 124         if (!$indexation)
  125. 125         {
  126. 126             $alias = new Alias(NULL, $string);
  127. 127             if (Validate::isLoadedObject($alias))
  128. 128                 $string = $alias->search;
  129. 129         }
  130. 130 
  131. 131         if ($indexation)
  132. 132         {
  133. 133             $minWordLen = (int)Configuration::get('PS_SEARCH_MINWORDLEN');
  134. 134             if ($minWordLen > 1)
  135. 135             {
  136. 136                 $minWordLen -= 1;
  137. 137                 $string = preg_replace('/(?<=\s)[^\s]{1,'.$minWordLen.'}(?=\s)/Su', ' ', $string);
  138. 138                 $string = preg_replace('/^[^\s]{1,'.$minWordLen.'}(?=\s)/Su', '', $string);
  139. 139                 $string = preg_replace('/(?<=\s)[^\s]{1,'.$minWordLen.'}$/Su', '', $string);
  140. 140                 $string = preg_replace('/^[^\s]{1,'.$minWordLen.'}$/Su', '', $string);
  141. 141             }
  142. 142         }
  143. 143 
  144. 144         $string = trim(preg_replace('/\s+/', ' ', $string));
  145. 145         return $string;
  146. 146     }

  472. 472                         foreach ($addedWords AS $wordId)
