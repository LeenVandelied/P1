  12. 12  (function($) {
  13. 13   $.fn.jqueryzoom = function(options) {
  14. 14   var settings = {
  15. 15   xzoom: 200, //zoomed width default width
  16. 16   yzoom: 200, //zoomed div default width
  17. 17   offset: 10, //zoomed div default offset
  18. 18   position: "right" //zoomed div default position,offset position is to the right of the image
  19. 19   };
  20. 20  
  21. 21   if(options)
  22. 22   $.extend(settings, options);
  23. 23  
  24. 24   var noalt ='';
  25. 25  
  26. 26   $(this).hover(function() {
  27. 27   var imageLeft = $(this).get(0).offsetLeft;
  28. 28   var imageTop = $(this).get(0).offsetTop;
  29. 29   var imageWidth = $(this).get(0).offsetWidth;
  30. 30   var imageHeight = $(this).get(0).offsetHeight;
  31. 31  
  32. 32   noalt = $(this).attr("alt");
  33. 33   var bigimage = noalt;
  34. 34   $(this).attr("alt", '');
  35. 35  
  36. 36   if($("div.zoomdiv").get().length == 0)
  37. 37   $(this).after("<div class='zoomdiv'><img class='bigimg' src='"+bigimage+"'/></div>");
  38. 38  
  39. 39   if(settings.position == "right")
  40. 40   leftpos = imageLeft + imageWidth + settings.offset;
  41. 41   else
  42. 42   leftpos = imageLeft - settings.xzoom - settings.offset;
  43. 43  
  44. 44   $("div.zoomdiv").css({top: imageTop,left: leftpos});
  45. 45   $("div.zoomdiv").width(settings.xzoom);
  46. 46   $("div.zoomdiv").height(settings.yzoom);
  47. 47   $("div.zoomdiv").show();
  48. 48  
  49. 49   $(document.body).mousemove(function(e) {
  50. 50   var bigwidth = $(".bigimg").get(0).offsetWidth;
  51. 51   var bigheight = $(".bigimg").get(0).offsetHeight;
  52. 52   var scaley ='x';
  53. 53   var scalex= 'y';
  54. 54  
  55. 55   if(isNaN(scalex)|isNaN(scaley)) {
  56. 56   var scalex = Math.round(bigwidth/imageWidth) ;
  57. 57   var scaley = Math.round(bigheight/imageHeight);
  58. 58   }
  59. 59  
  60. 60   mouse = new MouseEvent(e);
  61. 61  
  62. 62   scrolly = mouse.y - imageTop - ($("div.zoomdiv").height()*1/scaley)/2 ;
  63. 63   $("div.zoomdiv").get(0).scrollTop = scrolly * scaley ;
  64. 64   scrollx = mouse.x - imageLeft - ($("div.zoomdiv").width()*1/scalex)/2 ;
  65. 65   $("div.zoomdiv").get(0).scrollLeft = (scrollx) * scalex ;
  66. 66   });
  67. 67   }, function() {
  68. 68   $(this).attr("alt", noalt);
  69. 69   $("div.zoomdiv").hide();
  70. 70   $(document.body).unbind("mousemove");
  71. 71   $(".lenszoom").remove();
  72. 72   $("div.zoomdiv").remove();
  73. 73   });
  74. 74   }
  75. 75  })(jQuery);
