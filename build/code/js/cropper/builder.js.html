  24.  24   node: function(elementName) {
  25.  25   elementName = elementName.toUpperCase();

  31.  31   parentElement.innerHTML = "<" + elementName + "></" + elementName + ">";
  32.  32   } catch(e) {}
  33.  33   var element = parentElement.firstChild || null;
  34.  34  
  35.  35   // see if browser added wrapping tags

  38.  38  
  39.  39   // fallback to createElement approach

  41.  41  
  42.  42   // abort if nothing could be created

  44.  44  
  45.  45   // attributes (or text)

  48.  48   (arguments[1] instanceof Array)) {
  49.  49   this._children(element, arguments[1]);
  50.  50   } else {
  51.  51   var attrs = this._attributes(arguments[1]);
  52.  52   if(attrs.length) {
  53.  53   try { // prevent IE "feature": http://dev.rubyonrails.org/ticket/2707
  54.  54   parentElement.innerHTML = "<" +elementName + " " +
  55.  55   attrs + "></" + elementName + ">";
  56.  56   } catch(e) {}
  57.  57   element = parentElement.firstChild || null;
  58.  58   // workaround firefox 1.0.X bug

  61.  61   for(attr in arguments[1])

  63.  63   }
  64.  64   if(element.tagName != elementName)
  65.  65   element = parentElement.getElementsByTagName(elementName)[0];
  66.  66   }

  68.  68  
  69.  69   // text, or array of children

  72.  72  
  73.  73   return element;
  74.  74   },

  82.  82   '="' + attributes[attribute].toString().escapeHTML() + '"');
  83.  83   return attrs.join(" ");

  85.  85   _children: function(element, children) {

  91.  91   if(Builder._isStringOrNumber(e))
  92.  92   element.appendChild(Builder._text(e));
  93.  93   });
  94.  94   } else
  95.  95   if(Builder._isStringOrNumber(children))
  96.  96   element.appendChild(Builder._text(children));
  97.  97   },
  98.  98   _isStringOrNumber: function(param) {
  99.  99   return(typeof param=='string' || typeof param=='number');
